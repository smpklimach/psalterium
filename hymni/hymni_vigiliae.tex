\begin{multicols}{2}
 


\noindent\frot{P}ræco diéi iam sonat, \\
noctis profúndæ pérvigil, \\
noctúrna lux viántibus\\
a nocte noctem ségregans.\\

\noindent\frot{H}oc excitátus lúcifer\\
solvit polum calígine;\\
hoc omnis errónum chorus\\
vias nocéndi déserit.\\

\noindent\frot{H}oc nauta vires cólligit\\
pontíque mitéscunt freta;\\
hoc, ipse Petra Ecclésiæ, \\
canénte, culpam díluit.\\

\noindent\frot{I}esu, labántes réspice\\
et nos vidéndo córrige;\\
si réspicis, lapsus cadunt\\
fletúque culpa sólvitur.\\

\noindent\frot{T}u, lux, refúlge sénsibus\\
mentísque somnum díscute;\\
te nostra vox primum sonet\\
et vota solvámus tibi.\\

\noindent\frot{S}it, Christe, rex piíssime, \\
tibi Patríque glória\\
cum Spíritu Paráclito,\\ 
in sempitérna sæcula. Amen.\\

\end{multicols}
