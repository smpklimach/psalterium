\section[ad vigilias]{AD VIGILIAS}

\textcolor{red}{Invit.} Regem cui ómnia vivunt, * veníte adorémus. \rot{vj. T.}

\vspace{0.4cm}

\crot{Hymnus} 

\begin{multicols}{2}

\noindent{Q}ui vivis ante sæcula\\
vitæque fons es unicus,\\
nos, Deus, morti obnóxios\\
culpæque reos áspice.\\

\noindent{P}eccánti, Pater, hómini\\
pœnam sanxísti intéritum,\\
ut, pulvis datus púlveri,\\
se súbderet piáculum.\\

\noindent{V}itále sed spiráculum\\
quod indidísti próvidus,\\
æternitàtis pérmanet\\
germen immarcescíbile.\\

\noindent{H}æc spes, hoc est solácium:\\
revirescémus, Dómine,\\
primúsque resurgéntium\\
ad te nos Christus rápiet.\\

\noindent\rot{Pro mortuo:}\\
\noindent{H}oc vitæ regno pérfrui\\
défunctum præsta fámul-um,\\
quem Christi fides ímbuit,\\
quem almus unxit Spíritus.\\

\noindent\rot{Pro mortua:}\\
\noindent{H}oc vitæ regno pérfrui\\
défunctum præsta fámulam,\\
quam Christi fides ímbuit,\\
quam almus unxit Spíritus.\\

\noindent{H}oc regnum nobis própera\\
e terris cum cessérimus,\\
ut concinámus ómnium\\
te finem, te princípium.\\ Amen.

\end{multicols}
